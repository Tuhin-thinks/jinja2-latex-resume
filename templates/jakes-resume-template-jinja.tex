% Custom commands
\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}}
  }
}

\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeProjectHeading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubItem}[1]{\resumeItem{#1}\vspace{-4pt}}

\renewcommand\labelitemii{$\vcenter{\hbox{\tiny$\bullet$}}$}

\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=0.15in, label={}]}
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}
\newcommand{\resumeItemListStart}{\begin{itemize}}
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-5pt}}

%-----------EXPERIENCE-----------
\section*{ {{ headings.work | default("Work Experience") | latex_escape }} }
\resumeSubHeadingListStart
{% for job in work %}
  \resumeSubheading
    { {{ job.position | default('') | latex_escape }} }
    { {% if job.startDate %}{{ job.startDate | latex_escape }}{% endif %}{% if job.endDate %} -- {{ job.endDate | latex_escape }}{% endif %} }
    { {{ job.company | default('') | latex_escape }} }
    { {{ job.location | default('') | latex_escape }} }
    {% if job.highlights %}
      \resumeItemListStart
        {% for h in job.highlights %}
          \resumeItem{ {{ h | latex_escape }} }
        {% endfor %}
      \resumeItemListEnd
    {% endif %}
{% endfor %}
\resumeSubHeadingListEnd


%-----------PROJECTS-----------
{% if projects %}
\section*{ {{ headings.projects | default("Projects") | latex_escape }} }
\resumeSubHeadingListStart
{% for p in projects %}
  {% set left = "\\textbf{ " ~ (p.name | latex_escape) ~ " }" %}
  {% set right = "" %}
  {% if p.startDate is defined or p.endDate is defined %}
    {% set right = (p.startDate | default('') | latex_escape) ~ " -- " ~ (p.endDate | default('') | latex_escape) %}
  {% endif %}
  {% if p.keywords is defined and p.keywords %}
    {% set left = left ~ " $|$ \\emph{ " ~ (p.keywords | join(", ") | latex_escape) ~ " }" %}
  {% endif %}
  \resumeProjectHeading{ {{ left }} }{ {{ right }} }
  {% if p.description or p.url %}
    \resumeItemListStart
      {% if p.description %}
        {% if p.description is iterable and p.description is not string %}
          {% for desc_item in p.description %}
            \resumeItem{ {{ desc_item | latex_escape }} }
          {% endfor %}
        {% else %}
          \resumeItem{ {{ p.description | latex_escape }} }
        {% endif %}
      {% endif %}
      {% if p.url %}
        \resumeItem{ Project link:\href{ {{ p.url | latex_escape }} }{ \color{blue} {{ p.url | latex_escape }} }}
      {% endif %}
    \resumeItemListEnd
    {% endif %}
{% endfor %}
\resumeSubHeadingListEnd
{% endif %}
